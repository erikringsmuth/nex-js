Full site example

This site is built on RequireJS, nex-js, and the RequireJS Router. The site's source is available in the gh-pages branch of nex-js. This is a great example of a full site setup. Dependencies are installed using Bower.

Simple single page app example

/index.html

<!doctype html>
<html>
  <head>
    <title>nex-js</title>
    <script data-main="js/main" src="js/lib/require/require.js"></script>
  </head>
  <body>
    <!-- main.js bootstraps the javascript app and inserts it here -->
  </body>
</html>

/js/main.js

(function() {
  'use strict';

  require.config({
    paths: {
      'nex': 'lib/nex-js/nex',
      'text': 'lib/require/text',
      'handlebars': 'lib/handlebars/handlebars'
    },
    shim: {
      handlebars: {
        exports: 'Handlebars'
      }
    }
  });

  require(['example/exampleView'], function(ExampleView) {
    document.querySelector('body').appendChild(new ExampleView().render().outerEl);
  });
})();

/js/example/exampleView.js

define([
  'nex',
  'handlebars',
  'text!./exampleTemplate.html'
], function(Nex, Handlebars, exampleTemplate) {
  'use strict';

  return Nex.View.extend({
    template: Handlebars.compile(exampleTemplate),

    model: {name: 'Erik Ringsmuth'},

    events: {
      'keyup input': 'inputChangeHandler'
    },
    
    inputChangeHandler: function(event) {
      this.model.name = event.target.value;
      this.render();
    }
  });
});

/js/example/exampleTemplate.html

<div>
  <div>nex-js - Hello \{{model.name}}</div>
  <div><input type="text" value="{{model.name}}" /></div>
</div>
